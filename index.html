    /* ── SCORING ──────────────────────────────────────── */
    function computeScores() {
    if (deck.length === 0) return { total:0, nachhaltigkeit:0, stabilitaet:0, wirtschaftlichkeit:0, warnings:[] };
    const cards = deck.map(id => CARDS.find(c => c.id === id));
    const n = cards.length;
    let nach = cards.reduce((s,c) => s + c.score_nachhaltigkeit, 0) / n;
    let stab = cards.reduce((s,c) => s + c.score_stabilitaet, 0) / n;
    let wirt = cards.reduce((s,c) => s + c.score_wirtschaftlichkeit, 0) / n;
    const types = [...new Set(cards.map(c => c.type))];
    if (types.length >= 3) { nach += 5; stab += 5; wirt += 5; }
    if (types.length >= 4) { nach += 5; stab += 5; }
    const hasVariable = cards.some(c => c.variabel);
    const hasSpeicher = cards.some(c => c.speicher);
    const hasFlexible = cards.some(c => c.flexibel);
    if (hasVariable && hasSpeicher) stab += 15;
    if (hasVariable && hasFlexible) stab += 10;
    const eCards = cards.filter(c => c.type === 'erzeugung');
    const hasErzeugung = eCards.length > 0;
    if (hasErzeugung && eCards.every(c => c.variabel) && !hasSpeicher && !hasFlexible) stab -= 20;
    if (cards.find(c => c.id === 'gas')) nach -= 10;
    nach = Math.min(100, Math.max(0, Math.round(nach)));
    stab = Math.min(100, Math.max(0, Math.round(stab)));
    wirt = Math.min(100, Math.max(0, Math.round(wirt)));
    const total = Math.round(nach * 0.4 + stab * 0.35 + wirt * 0.25);
    const warnings = [];
    const allVarE = hasErzeugung && eCards.every(c => c.variabel);
    const cm = (c, a) => `color-mix(in srgb, ${c} ${a}%, transparent)`;
    if (allVarE && !hasSpeicher)
    warnings.push({ icon:"⚠️", text:"Nur variable Erzeuger ohne Speicher – Dunkelflauten-Risiko! Bis 2030 werden 50–100 GW Speicher benötigt.", color:"var(--solar)", bg:"var(--solar-bg)", bc:cm("var(--solar)",30) });
    if (cards.find(c => c.id === 'gas') && !cards.some(c => c.id !== 'gas' && c.type === 'erzeugung'))
    warnings.push({ icon:"🔥", text:"Nur fossile Erzeugung – hohe CO₂-Emissionen (~490 g/kWh). Risiko eines fossilen Lock-In.", color:"var(--red)", bg:"var(--red-bg)", bc:cm("var(--red)",30) });
    if (!cards.some(c => c.type === 'waerme') && n >= 3)
    warnings.push({ icon:"💡", text:"Keine Wärmetechnologie – Wärme macht >50% des Energieverbrauchs aus. Die Wärmewende ist die größte ungelöste Aufgabe.", color:"var(--wind)", bg:"var(--wind-bg)", bc:cm("var(--wind)",30) });
    if (types.length >= 3 && hasSpeicher && hasVariable)
    warnings.push({ icon:"🌱", text:"Diverser Mix mit Speicherabsicherung – auf dem Weg zum 80%-EE-Ziel 2030!", color:"var(--growth)", bg:"var(--growth-bg)", bc:cm("var(--growth)",30) });
    if (cards.some(c => c.id === 'suffizienz'))
    warnings.push({ icon:"♻️", text:"Suffizienz im Mix! Die sauberste Energie ist die, die gar nicht erst verbraucht wird.", color:"var(--purple)", bg:"var(--purple-bg)", bc:cm("var(--purple)",30) });
    return { total, nachhaltigkeit:nach, stabilitaet:stab, wirtschaftlichkeit:wirt, warnings };
    }

    function trilemmaPoint(score, vx, vy, cx, cy) {
    const t = score / 100;
    return [cx + t * (vx - cx), cy + t * (vy - cy)];
    }

    function co2Params(co2) {
    if (co2 < 0) return { label:`${Math.abs(co2)} g gespart`, color:"var(--growth)", bg:"var(--growth-bg)" };
    if (co2 === 0) return { label:"0 g direkt", color:"var(--growth)", bg:"var(--growth-bg)" };
    if (co2 < 50) return { label:`${co2} g`, color:"var(--growth)", bg:"var(--growth-bg)" };
    if (co2 < 250) return { label:`${co2} g`, color:"var(--solar)", bg:"var(--solar-bg)" };
    return { label:`${co2} g`, color:"var(--red)", bg:"var(--red-bg)" };
    }
    /* ── RENDER ──────────────────────────────────────── */
    function render() {
    const scores = computeScores();
    const app = document.getElementById('app');
    const totalColor = scores.total >= 70 ? 'var(--growth)' : scores.total >= 40 ? 'var(--solar)' : 'var(--red)';
    const triSvg = buildTrilemma(scores.nachhaltigkeit, scores.stabilitaet, scores.wirtschaftlichkeit);
    /* Deck slots */
    let slotsHtml = '';
    for (let i = 0; i < MAX_DECK; i++) {
    if (i < deck.length) {
    const card = CARDS.find(c => c.id === deck[i]);
    const tm = TYPE_META[card.type];
    slotsHtml += `<div class="deck-slot filled" style="background:${tm.bg}; border-color:${tm.color}50;" onclick="removeFromDeck('${card.id}')">
    <span class="slot-remove">×</span>
    <span class="slot-icon">${card.icon}</span>
    <span class="slot-name">${card.name}</span>
    </div>`;
    } else {
    slotsHtml += `<div class="deck-slot empty"></div>`;
    }
    }
    /* Warnings */
    const warningsHtml = scores.warnings.map(w =>
    `<div class="warning" style="background:${w.bg}; color:${w.color}; border-color:${w.bc};">
    <span class="warn-icon">${w.icon}</span><span>${w.text}</span>
    </div>`
    ).join('');
    /* Cards */
    const sections = [
    { type:'erzeugung', label:'Stromerzeugung' },
    { type:'speicher', label:'Speicher' },
    { type:'waerme', label:'Wärme & Effizienz' },
    { type:'suffizienz', label:'Suffizienz' },
    ];
    let cardsHtml = '';
    for (const sec of sections) {
    const tm = TYPE_META[sec.type];
    cardsHtml += `
    <div class="cat-header">
    <div class="cat-line"></div>
    <div class="cat-badge" style="color:${tm.color}; background:${tm.bg}; border-color:${tm.color}30;">${sec.label}</div>
    <div class="cat-line"></div>
    </div>
    <div class="card-grid">`;
    for (const card of CARDS.filter(c => c.type === sec.type)) {
    const inDeck = deck.includes(card.id);
    const isExpanded = expandedCard === card.id;
    const co2 = co2Params(card.co2_lifecycle);
    const tm2 = TYPE_META[card.type];
    let detailHtml = '';
    if (isExpanded) {
    detailHtml = `<div class="card-detail">
    <p class="card-desc">${card.desc}</p>
    <div class="data-grid">
    <div class="data-cell">
    <div class="data-cell-label">Leistung</div>
    <div class="data-cell-val">${card.leistung}</div>
    </div>
    <div class="data-cell">
    <div class="data-cell-label">Volllaststunden</div>
    <div class="data-cell-val">${card.volllaststunden}</div>
    </div>
    <div class="data-cell">
    <div class="data-cell-label">LCOE 2024</div>
    <div class="data-cell-val">${card.lcoe_min > 0 ? card.lcoe_min+'–'+card.lcoe_max+' '+card.lcoe_unit : 'keine Erzeugung'}</div>
    </div>
    <div class="data-cell">
    <div class="data-cell-label">CO₂ Lifecycle</div>
    <div class="data-cell-val" style="color:${co2.color};">${co2.label}/kWh</div>
    </div>
    </div>
    <div class="card-source">📎 ${card.source}</div>
    <button class="add-btn" onclick="event.stopPropagation(); addToDeck('${card.id}')" ${inDeck || deck.length >= MAX_DECK ? 'disabled' : ''}>
    ${inDeck ? 'Bereits im Energiemix' : deck.length >= MAX_DECK ? 'Mix voll (max. 6)' : '+ Zum Energiemix hinzufügen'}
    </button>
    </div>`;
    }
    cardsHtml += `<div class="tech-card ${inDeck?'in-deck':''} ${isExpanded?'expanded':''}" onclick="toggleExpand('${card.id}')" style="--card-color:${tm2.color};">
    <div style="height:3px; background:${tm2.color}; opacity:${inDeck?0.2:1};"></div>
    <div class="card-inner">
    <div class="card-top">
    <span class="card-icon">${card.icon}</span>
    <span class="card-badge" style="color:${tm2.color}; background:${tm2.bg}; border-color:${tm2.color}30;">${tm2.label}</span>
    </div>
    <div class="card-name">${card.name}</div>
    <div class="card-subtitle">${card.leistung}${card.lcoe_min > 0 ? ' · '+card.lcoe_min+'–'+card.lcoe_max+' '+card.lcoe_unit : ''}</div>
    <div class="card-chips">
    <span class="chip" style="background:${co2.bg}; color:${co2.color};">CO₂ ${co2.label}</span>
    ${card.flexibel ? `<span class="chip" style="background:var(--wind-bg); color:var(--wind);">steuerbar</span>` : ''}
    ${card.variabel ? `<span class="chip" style="background:var(--purple-bg); color:var(--purple);">variabel</span>` : ''}
    ${card.speicher ? `<span class="chip" style="background:var(--wind-bg); color:var(--wind);">Speicher</span>` : ''}
    </div>
    </div>
    ${detailHtml}
    </div>`;
    }
    cardsHtml += `</div>`;
    }
    app.innerHTML = `
    <div class="header">
    <div class="header-eyebrow">Interaktiver Energieplaner</div>
    <h1>Energiewende<span>Baukasten</span></h1>
    <p class="header-sub">Kombiniere bis zu 6 Technologien zu deinem persönlichen Energiemix.<br>Alle Daten aus Forschung und Praxis.</p>
    </div>
    <div class="score-panel">
    <div class="trilemma-svg-wrap">
    ${triSvg}
    <div class="tri-labels">
    <div class="tri-label-row"><span class="tri-dot" style="background:var(--growth);"></span>Nachhaltigkeit</div>
    <div class="tri-label-row"><span class="tri-dot" style="background:var(--wind);"></span>Stabilität</div>
    <div class="tri-label-row"><span class="tri-dot" style="background:var(--solar);"></span>Wirtschaftlichk.</div>
    </div>
    </div>
    <div class="score-right">
    <div>
    <div class="score-label">Gesamtscore</div>
    <div class="score-total-row">
    <div class="score-number" style="color:${deck.length > 0 ? totalColor : 'var(--text-muted)'};">${deck.length > 0 ? scores.total : '–'}</div>
    <div class="score-denom">/100</div>
    </div>
    </div>
    <div class="tri-bars">
    <div class="tri-bar-row">
    <div class="tri-bar-name">Nachh.</div>
    <div class="tri-track"><div class="tri-fill" style="width:${scores.nachhaltigkeit}%; background:var(--growth);"></div></div>
    <div class="tri-bar-val" style="color:var(--growth);">${scores.nachhaltigkeit}</div>
    </div>
    <div class="tri-bar-row">
    <div class="tri-bar-name">Stab.</div>
    <div class="tri-track"><div class="tri-fill" style="width:${scores.stabilitaet}%; background:var(--wind);"></div></div>
    <div class="tri-bar-val" style="color:var(--wind);">${scores.stabilitaet}</div>
    </div>
    <div class="tri-bar-row">
    <div class="tri-bar-name">Wirt.</div>
    <div class="tri-track"><div class="tri-fill" style="width:${scores.wirtschaftlichkeit}%; background:var(--solar);"></div></div>
    <div class="tri-bar-val" style="color:var(--solar);">${scores.wirtschaftlichkeit}</div>
    </div>
    </div>
    <div class="deck-count">${deck.length} / ${MAX_DECK} Technologien · Karte tippen zum Entfernen</div>
    </div>
    </div>
    <div class="deck-section">
    <div class="section-header">
    <div class="section-eyebrow">Dein Energiemix</div>
    </div>
    <div class="deck-slots">${slotsHtml}</div>
    </div>
    <div class="warnings-area">${warningsHtml}</div>
    ${cardsHtml}
    <div class="info-box" style="margin-top:32px;">
    <div class="info-box-title">Datenquellen</div>
    <p>
    Alle Kennzahlen basieren auf dem <strong>Spickzettel Energiewende</strong> sowie wissenschaftlichen Quellen:
    <strong>LCOE</strong>: Fraunhofer ISE Juli 2024.
    <strong>CO₂ Lifecycle</strong>: IPCC AR5 Medianwerte (gCO₂eq/kWh).
    <strong>Leistung & Verfügbarkeit</strong>: Spickzettel (Praxiswerte).
    Das Scoring bildet das <strong>Energie-Trilemma</strong> ab: Nachhaltigkeit (40 %), Stabilität (35 %), Wirtschaftlichkeit (25 %) – plus Boni für Diversität und Speicherabsicherung.
    </p>
    <p class="src">Fraunhofer ISE (2024): Stromgestehungskosten; IPCC (2014): AR5 WGIII Annex III; Spickzettel Energiewende</p>
    </div>
    <div class="footer">
    Basierend auf dem Spickzettel Energiewende<br>
    <a href="https://www.ise.fraunhofer.de" target="_blank">Fraunhofer ISE</a> &nbsp;·&nbsp;
    <a href="https://www.ipcc.ch" target="_blank">IPCC</a> &nbsp;·&nbsp;
    <a href="https://www.umweltbundesamt.de" target="_blank">UBA</a>
    </div>
    `;
    }
    render();
    </script>
